async function runAudit(e){
  e.preventDefault();
  const url = document.querySelector('#url').value.trim();
  const email = document.querySelector('#email').value.trim();
  const name = document.querySelector('#name').value.trim();
  if(!/^https?:\/\//.test(url)) return toast('Enter a full URL incl. https://');
  btn.disabled = true; showSpinner();

  const r = await fetch(import.meta.env.VITE_API_BASE + '/api/audit', {
    method:'POST',
    headers:{'Content-Type':'application/json'},
    body: JSON.stringify({ url, email, name })
  });
  const data = await r.json();
  if(!r.ok) throw new Error(data.error || 'audit_failed');

  // show results
  renderSummary({
    score: data.overallScore,
    fixes: data.topFixes,
    html: data.reportUrlHTML,
    pdf: data.reportUrlPDF
  });
}
